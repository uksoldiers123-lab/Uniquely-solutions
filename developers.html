Server: create a payment

// Node.js (example)
import express from 'express';
import Stripe from 'stripe';
const app = express();
app.use(express.json());

const stripe = new Stripe(process.env.STRIPE_SECRET);

// Minimal PaymentIntent for $20.00 USD
app.post('/api/intents', async (req, res) => {
  const { amount = 2000, currency = 'usd', metadata = {} } = req.body || {};
  const intent = await stripe.paymentIntents.create({
    amount, currency, metadata,
    automatic_payment_methods: { enabled: true }
  });
  res.json({ clientSecret: intent.client_secret });
});

app.listen(3000);

Authentication

Authenticate server‑side with your secret key using the Authorization: Bearer header. Never expose secret keys in client code.

# cURL pattern
curl https://api.example-payments.com/v1/charges \
  -u sk_test_xxx: \
  -d amount=2000 \
  -d currency=usd

Rotate keys regularly and scope them per environment (test vs live).

Checkout session (hosted)

// Create hosted checkout session
const session = await payments.checkout.create({
  mode: 'payment',
  success_url: 'https://yourapp.com/success',
  cancel_url: 'https://yourapp.com/cancel',
  line_items: [{ name: 'Sample', amount: 2000, currency: 'usd', quantity: 1 }],
  metadata: { orderId: 'ORD-1001' }
});

Hosted checkout surfaces wallets automatically on supported devices.
